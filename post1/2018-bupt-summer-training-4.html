A.CodeForces 791D

首先，先考虑一个问题：如何计算一棵树中所有点对之间的距离和。对于任意一个点，我们可以发现，如果这个点的子树的大小为$$s$$的话，那么这个点与其祖先之间的那条边在计算总和的过程中一定会被计算$$(n-s)\times s$$次（n为总的节点数）。但是目前有一个问题，就是每次只能走k步，比如说k=4，而距离为10，那就需要3步，此时如果直接求出总的距离之和去除以k，答案肯定是错误的。因此我们在dfs到每个节点的时候，还要计算他的各个子节点之间的距离，然后把所有mod k有余数的距离与k的倍数的差值都给补上。最后距离总和除以k即可。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 2e5 + 50;
ll n,k,ans=0,size[maxn][5],cntsub[maxn];//size代表节点x的距离它的距离为i的子节点的个数,cnt则为节点想的子树个数和
vector<int>G[maxn];
void dfs(ll x,ll fa,ll sit)//x为节点ID,fa为节点祖先,sit为当前节点到顶点的距离mod k
{
    cntsub[x] = size[x][sit % k] = 1;
    ll i, j;
    for (auto a : G[x]) {
        if (a == fa)continue;
        dfs(a, x, sit + 1);
        for (i = 0; i < k; i++) {//枚举到子节点的距离
            for (j = 0; j < k; j++) {
                ll t = (k - (i + j - 2 * (sit % k)) % k) % k;
                ans += t * size[x][i] * size[a][j];
            }
        }
        for (i = 0; i < k; i++)size[x][i] += size[a][i];
        cntsub[x] += cntsub[a];
    }
    ans += cntsub[x] * (n - cntsub[x]);//x与祖先连边被计算的次数
}
int main()
{
    int i, j, u, v;
    cin >> n >> k;
    for (i = 1; i < n; i++) {
        scanf("%d%d", &u, &v);
        G[u].push_back(v);
        G[v].push_back(u);
    }
    dfs(1, 0, 0);
    cout << ans / k << endl;

    return 0;
}

B.CodeForces - 2B

求矩阵路径上数字的积末尾最少的0个数。那么其实就是统计路径上最少的2个数与5个数，取其中比较小的即可。有个问题就是路径中如果有0，那么就需要特判答案到底是1还是0了。路径输出的时候稍微要点小技巧，没了。

#include<bits/stdc++.h>
using namespace std;
int get2(int x)
{
    int cnt=0;
    while(x&&x%2==0){
        cnt++;x/=2;
    }
    return cnt;
}
int get5(int x)
{
    int cnt=0;
    while(x&&x%5==0){
        cnt++;x/=5;
    }
    return cnt;
}
int num[1005][1005];
typedef pair<int,char>P;
P dp2[1005][1005],dp5[1005][1005];
int main()
{
    int i,j,k,n;
    cin>>n;
    memset(dp2,0x3f, sizeof(dp2));memset(dp5,0x3f,sizeof(dp5));
    int move1[2][2]={1,0,0,1};
    bool haszero=false;int zerox,zeroy;
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++) {
            scanf("%d", &num[i][j]);
            if(!num[i][j]){
                haszero=true;zerox=i,zeroy=j;
            }
        }
    if(num[1][1]) {
        dp2[1][1].first = get2(num[1][1]), dp5[1][1].first = get5(num[1][1]);
    }
    else{
        dp2[1][1].first=dp5[1][1].first=1;
    }
    for(i=1;i<=n;i++)
        for(j=1;j<=n;j++){
        //if(i==1&&j==1)continue;
        for(k=0;k<2;k++){
            int a=i+move1[k][0],b=j+move1[k][1];
            if(num[a][b]==0){
                dp2[a][b].first=0;dp5[a][b].first=0;
            }
            else {
                if (dp2[a][b].first > dp2[i][j].first + get2(num[a][b])) {
                    dp2[a][b].first = dp2[i][j].first + get2(num[a][b]);
                    if (k)dp2[a][b].second = 'R';
                    else dp2[a][b].second = 'D';
                }
                if (dp5[a][b].first > dp5[i][j].first + get5(num[a][b])) {
                    dp5[a][b].first = dp5[i][j].first + get5(num[a][b]);
                    if (k)dp5[a][b].second = 'R';
                    else dp5[a][b].second = 'D';
                }
            }
        }
    }
    int ans=1<<30;
    int sit=0;
    //cout<<dp2[n][n].first<<' '<<dp5[n][n].first<<endl;
    if(haszero&&dp2[n][n].first>1&&dp5[n][n].first>1){
        ans=1;sit=1;
    }
    else{
        if(dp2[n][n].first>dp5[n][n].first){
            ans=dp5[n][n].first;sit=3;
        }
        else{
            ans=dp2[n][n].first;sit=2;
        }
    }
    cout<<ans<<endl;
    vector<char>way;
    if(sit==1){
        for(i=1;i<zerox;i++)way.push_back('D');
        for(j=1;j<n;j++)way.push_back('R');
        for(i=zerox;i<n;i++)way.push_back('D');
    }
    else if(sit==2){
        i=j=n;
        while(1){
            if(i==1&&j==1)break;
            way.push_back(dp2[i][j].second);
            if(dp2[i][j].second=='R')j--;
            else i--;
        }
        reverse(way.begin(),way.end());
    }
    else if(sit==3){
        i=j=n;
        while(1){
            if(i==1&&j==1)break;
            way.push_back(dp5[i][j].second);
            if(dp5[i][j].second=='R')j--;
            else i--;
        }
        reverse(way.begin(),way.end());
    }
    for(auto a:way)
        cout<<a;

    return 0;
}

D.CodeForces - 785B

主要考察英语阅读理解能力，注意先上哪个课程是不确定的。傻逼题* 2

#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,m,ans,i,j,a,b,amin=0,amax=1<<30,bmin=0,bmax=1<<30;
    cin>>n;
    for(i=1;i<=n;i++) {
        scanf("%d%d",&a,&b);
        amin = max(amin, a);
        amax = min(amax, b);
    }
    cin>>m;
    for(i=1;i<=m;i++) {
        scanf("%d%d",&a,&b);
        bmin = max(bmin, a);
        bmax = min(bmax, b);
    }
    ans=max(amin-bmax,bmin-amax);
    cout<<(ans>=0?ans:0)<<endl;

    return 0;
}

F.CodeForces - 57C

首先，注意到一点：每一个不下降数列反过来就是一个不上升数列，因此我们只要计算出不下降数列的个数，乘2，减去n（所有数字相同的数列个数），就是答案了。 然后，第一位肯定是要放一个数的，我们姑且不管。那么在后面的n-1个位置上，我们可以任意取若干个“上升点”，也就是这个点放的数比前面的数要来的大，相邻“上升点”之间的数字的值是相同的。在n-1个位置中选k个上升点，方案数是$$C_{n-1}^{k}$$,然后，选定k个上升点，那么我们这个数列中就需要k+1个不同的数（还记得第一位么？），会发现，任意选择1~n中的k+1个数，他都对应着唯一的一种排列方案，所以，总的答案就是$$\sum_{i=0}^{n-1}C_{n-1}^{i}\times C_n^{i+1}$$,算算组合数算算逆元就ok了。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int mod=1e9+7;
const int maxn=1e5+10;
ll c1[maxn],inv[maxn],jc[maxn],c2[maxn];
ll pow(ll x,ll n,ll mod)
{
    ll ans=1;
    x%=mod;
    while(n){
        if(n&1)
            ans=ans*x%mod;
        x=x*x%mod;
        n>>=1;
    }
    return ans;
}
int main()
{
    int i,j,k,n;
    cin>>n;
    jc[1]=1;inv[1]=1,inv[0]=1;
    for(i=2;i<=n;i++)
        jc[i]=jc[i-1]*i%mod,inv[i]=pow(jc[i],mod-2,mod);
    for(i=1;i<=n;i++) {
        c1[i] = (jc[n - 1] * inv[i] % mod * inv[n - 1 - i]) % mod;
        c2[i] = (jc[n] * inv[i] % mod * inv[n - i]) % mod;
    }
    c1[0]=1;c2[0]=1;
    ll ans=0;
    for(i=0;i<n;i++)
        ans=(ans+c1[i]*c2[i+1])%mod;
    ans=(ans+ans-n)%mod;
    cout<<ans<<endl;

    return 0;
}

J.CodeForces - 145C

给出一个长度为n的序列ai，问其长度为k的子序列中，同一个幸运数字至多出现一次的子序列个数，幸运数字即为全部由4和7组成的数字

假设有$$res$$个不同的幸运数字，有$$sum$$个非幸运数字，第$$i$$个幸运数字在原序列中出现了$$num[i]$$次，用$$dp[i][j]$$表示从前$$i$$个幸运数字中选$$j$$个的方案数，那么易得转移方程$$dp[i][j]=dp[i−1][j]+num[i]⋅dp[i−1][j−1]$$，之后枚举子序列中非幸运数字个数即可得到答案$$ans=\sum_{i=0}^{min(k,sum)}C_{sum}^{i}\times dp[res][k-i]$$。

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn=1e5+10;
const int mod=1e9+7;
ll fac[maxn],dp[3000][3000];
bool islucky(int v)//判断是否lucky number
{
    while (v) {
        if (v % 10 != 4 && v % 10 != 7)
            return false;
        v /= 10;
    }
    return true;
}
ll pow(ll x,ll n,ll mod)
{
    ll ans=1;
    x%=mod;
    while(n){
        if(n&1)
            ans=ans*x%mod;
        x=x*x%mod;
        n>>=1;
    }
    return ans;
}
ll C(ll v, ll u)
{
    if (v < u)
        return 0;
    ll x = (fac[u] * fac[v - u]) % mod;
    return (fac[v] * pow(x, mod - 2,mod)) % mod;
}
int main()
{
    int i, j, n, k,sum=0;ll ans=0;
    cin>>n>>k;
    map<int,int>mp1;
    fac[0] = 1;
    for (i = 1; i <= n; i++)
        fac[i] = (fac[i - 1] * i) % mod;
    for (i = 1; i <= n; i++) {
        scanf("%d", &j);
        if (islucky(j))
            mp1[j]++;//统计幸运数个数
        else sum++;
    }
    dp[0][0] = 1;i = 1;
    for (auto it = mp1.begin(); it != mp1.end(); it++, i++)
        for (j = 0; j < i; j++) {
            dp[i][j] = (dp[i][j] + dp[i - 1][j]) % mod;
            dp[i][j + 1] = (dp[i][j + 1] + dp[i - 1][j] * it->second) % mod;
        }
    int cnt = mp1.size();
    for (i = 0; i <= cnt && i <= k; i++)
        ans = (ans + (dp[cnt][i] * C(sum, k - i)) % mod) % mod;
    cout<<ans<<endl;

    return 0;
}

K.CodeForces - 691B

难点在于如何找出所有对称的字符，傻逼题* 4

#include<bits/stdc++.h>
using namespace std;
bool check(char x, char y)
{
    if (x == 'A' && y == 'A'||x == 'H' && y == 'H'||x == 'I' && y == 'I'||x == 'M' && y == 'M'||x == 'O' && y == 'O')
        return true;
    if (x == 'T' && y == 'T'||x == 'U' && y == 'U'||x == 'V' && y == 'V'||x == 'W' && y == 'W'||x == 'X' && y == 'X')
        return true;
    if (x == 'Y' && y == 'Y'||x == 'v' && y == 'v'||x == 'w' && y == 'w'||x == 'x' && y == 'x'||x == 'o' && y == 'o')
        return true;
    if (x == 'p' && y == 'q'||x == 'q' && y == 'p'||x == 'b' && y == 'd'||x=='d'&&y=='b')
        return true;
    return false;
}
int main()
{
    string str1,str2;
    cin>>str1;str2=str1;
    reverse(str2.begin(),str2.end());
    for(int i=0;i<str1.size();i++){
        if(!check(str1[i],str2[i])){
            cout<<"NIE"<<endl;return 0;
        }
    }
    cout<<"TAK"<<endl;

    return 0;
}
